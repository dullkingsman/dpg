FUNCTION public.get_user(user_id bigint)
    RETURNS text
    LANGUAGE plpgsql
    STABLE
    SECURITY DEFINER
    AS $$
BEGIN
    RETURN (SELECT email FROM users WHERE id = user_id);
END;
$$
{
    COMMENT "fetch user email by id";
}
